```sh
$> pip3 install virtualenv

$ brew info python@3.10

$ virtualenv venv -p python3.10
$ source ./venv/bin/activate
$ python3 --version && pip3 --version

$ pip3 install -U pip && pip3 install rasa
$ rasa --version

$ rasa init

# https://rasa.com/docs/rasa/chitchat-faqs

$ rasa train

$ rasa shell
$ rasa run

...
$ deactivate
```
